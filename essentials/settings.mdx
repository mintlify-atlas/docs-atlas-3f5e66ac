Every Mintlify site needs a `docs.json` file with the core configuration settings. This file controls everything from your site's branding to navigation structure, theme colors, and integrations.

## Properties

### Basic Configuration

<ResponseField name="name" type="string" required>
  Name of your project. Used for the global title and displayed in the browser tab.

  ```json
  "name": "Mint Starter Kit"
  ```
</ResponseField>

<ResponseField name="$schema" type="string">
  JSON schema reference for docs.json validation and autocomplete in your editor.

  ```json
  "$schema": "https://mintlify.com/docs.json"
  ```
</ResponseField>

### Branding

<ResponseField name="logo" type="string | object">
  Path to logo image or object with paths for light and dark mode logos.
  
  <Expandable title="Logo Configuration">
    <ResponseField name="light" type="string">
      Path to the logo in light mode
      
      ```json
      "light": "/logo/light.svg"
      ```
    </ResponseField>
    
    <ResponseField name="dark" type="string">
      Path to the logo in dark mode
      
      ```json
      "dark": "/logo/dark.svg"
      ```
    </ResponseField>
    
    <ResponseField name="href" type="string" default="/">
      Where clicking on the logo links you to
    </ResponseField>
  </Expandable>

  <CodeGroup>
    ```json Simple Logo
    "logo": "/logo.svg"
    ```

    ```json Light/Dark Mode Logos
    "logo": {
      "light": "/logo/light.svg",
      "dark": "/logo/dark.svg"
    }
    ```
  </CodeGroup>
</ResponseField>

<ResponseField name="favicon" type="string">
  Path to the favicon image displayed in the browser tab.

  ```json
  "favicon": "/favicon.svg"
  ```
</ResponseField>

### Theme & Colors

<ResponseField name="theme" type="string">
  The color theme preset for your documentation site.

  ```json
  "theme": "mint"
  ```
</ResponseField>

<ResponseField name="colors" type="object" required>
  Hex color codes for your global theme. These colors are used throughout your documentation for accents, buttons, links, and more.
  
  <Expandable title="Color Properties">
    <ResponseField name="primary" type="string" required>
      The primary color used for highlighted content, section headers, and accents in light mode.
      
      ```json
      "primary": "#16A34A"
      ```
    </ResponseField>
    
    <ResponseField name="light" type="string">
      The primary color for dark mode accents and highlights.
      
      ```json
      "light": "#07C983"
      ```
    </ResponseField>
    
    <ResponseField name="dark" type="string">
      The primary color for important buttons and calls-to-action.
      
      ```json
      "dark": "#15803D"
      ```
    </ResponseField>
    
    <ResponseField name="background" type="object">
      Custom background colors for light and dark mode.
      
      <Expandable title="Background Object">
        <ResponseField name="light" type="string" required>
          The hex color code of the background in light mode
        </ResponseField>
        <ResponseField name="dark" type="string" required>
          The hex color code of the background in dark mode
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>

  ```json Example from Mint Starter Kit
  "colors": {
    "primary": "#16A34A",
    "light": "#07C983",
    "dark": "#15803D"
  }
  ```
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  A background image to be displayed behind every page. The image should be optimized for web use.

  ```json
  "backgroundImage": "/background.png"
  ```

  <Note>
    See live examples at [Infisical](https://infisical.com/docs) and [FRPC](https://frpc.io)
  </Note>
</ResponseField>

<ResponseField name="modeToggle" type="object">
  Customize the dark mode toggle behavior.
  
  <Expandable title="Mode Toggle Options">
    <ResponseField name="default" type="'light' | 'dark'">
      Set if you always want to show light or dark mode for new users. When not set, defaults to the user's operating system preference.
    </ResponseField>
    
    <ResponseField name="isHidden" type="boolean" default={false}>
      Set to `true` to hide the dark/light mode toggle and force a single mode.
    </ResponseField>
  </Expandable>

  <CodeGroup>
    ```json Force Dark Mode Only
    "modeToggle": {
      "default": "dark",
      "isHidden": true
    }
    ```

    ```json Force Light Mode Only
    "modeToggle": {
      "default": "light",
      "isHidden": true
    }
    ```
  </CodeGroup>
</ResponseField>

### Navigation Bar

<ResponseField name="navbar" type="object">
  Configure links and buttons in the top navigation bar.
  
  <Expandable title="Navbar Configuration">
    <ResponseField name="links" type="array">
      Array of link objects to display in the navbar.
      
      <Expandable title="Link Properties">
        <ResponseField name="label" type="string">
          The text displayed for the link
        </ResponseField>
        <ResponseField name="href" type="string">
          The URL the link points to
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="primary" type="object">
      Primary call-to-action button in the navbar.
      
      <Expandable title="Primary Button">
        <ResponseField name="type" type="'button' | 'link' | 'github'">
          The button style. `github` displays repo stars.
        </ResponseField>
        <ResponseField name="label" type="string">
          The button text
        </ResponseField>
        <ResponseField name="href" type="string">
          The button destination URL
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>

  ```json Example from Mint Starter Kit
  "navbar": {
    "links": [
      {
        "label": "Support",
        "href": "mailto:hi@mintlify.com"
      }
    ],
    "primary": {
      "type": "button",
      "label": "Dashboard",
      "href": "https://dashboard.mintlify.com"
    }
  }
  ```
</ResponseField>

<ResponseField name="topbarLinks" type="array">
  <Warning>Deprecated: Use `navbar.links` instead for the v2 navigation format.</Warning>
  
  Array of `name`s and `url`s of links in the topbar (legacy format).
  
  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">
      The name of the button
    </ResponseField>
    <ResponseField name="url" type="string">
      The url once you click on the button
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="object">
  <Warning>Deprecated: Use `navbar.primary` instead for the v2 navigation format.</Warning>
  
  Call-to-action button in the topbar (legacy format).
  
  <Expandable title="Topbar CTA">
    <ResponseField name="type" type="'link' | 'github'" default="link">
      `link` shows a button. `github` shows repo information including stars.
    </ResponseField>
    <ResponseField name="url" type="string">
      Button destination or GitHub repository URL
    </ResponseField>
    <ResponseField name="name" type="string">
      Text inside the button (required if `type` is `link`)
    </ResponseField>
  </Expandable>
</ResponseField>

### Footer

<ResponseField name="footer" type="object">
  Configure footer content and social links.
  
  <Expandable title="Footer Configuration">
    <ResponseField name="socials" type="object">
      Social media links displayed in the footer.
      
      Supported platforms: `website`, `facebook`, `x`, `discord`, `slack`, `github`, `linkedin`, `instagram`, `hacker-news`
      
      ```json
      "socials": {
        "x": "https://x.com/mintlify",
        "github": "https://github.com/mintlify",
        "linkedin": "https://linkedin.com/company/mintlify"
      }
      ```
    </ResponseField>
  </Expandable>

  ```json Example from Mint Starter Kit
  "footer": {
    "socials": {
      "x": "https://x.com/mintlify",
      "github": "https://github.com/mintlify",
      "linkedin": "https://linkedin.com/company/mintlify"
    }
  }
  ```
</ResponseField>

<ResponseField name="footerSocials" type="object">
  <Warning>Deprecated: Use `footer.socials` instead for the v2 navigation format.</Warning>
  
  Social media accounts in footer (legacy format).
</ResponseField>

### Navigation

<ResponseField name="navigation" type="object" required>
  Defines the structure and organization of your documentation pages. See the dedicated [Navigation](/essentials/navigation) page for detailed documentation.

  ```json Basic Structure
  "navigation": {
    "tabs": [...],
    "global": {...}
  }
  ```
</ResponseField>

<ResponseField name="tabs" type="array">
  <Warning>Deprecated: Use `navigation.tabs` instead for the v2 navigation format.</Warning>
  
  An array of navigational tabs (legacy format).
</ResponseField>

<ResponseField name="anchors" type="array">
  <Warning>Deprecated: Use `navigation.global.anchors` instead for the v2 navigation format.</Warning>
  
  An array of anchors with icons, colors, and URLs (legacy format).
</ResponseField>

<ResponseField name="topAnchor" type="object">
  Override the default configurations for the top-most anchor.
  
  <Expandable title="Top Anchor">
    <ResponseField name="name" type="string" default="Documentation">
      The name of the top-most anchor
    </ResponseField>
    <ResponseField name="icon" type="string" default="book-open">
      Font Awesome icon name
    </ResponseField>
    <ResponseField name="iconType" type="string" default="duotone">
      One of: "brands", "duotone", "light", "sharp-solid", "solid", or "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

### API Configuration

<ResponseField name="api" type="object">
  Configuration for API settings and playground.
  
  <Expandable title="API Settings">
    <ResponseField name="baseUrl" type="string | string[]">
      The base URL for all API endpoints. Can be an array to enable multiple base URL options.
    </ResponseField>

    <ResponseField name="auth" type="object">
      Authentication configuration for API playground.
      
      <Expandable title="Auth">
        <ResponseField name="method" type="'bearer' | 'basic' | 'key'">
          The authentication strategy used for all API endpoints
        </ResponseField>
        <ResponseField name="name" type="string">
          The name of the authentication parameter. For `basic` auth, format: `[usernameName]:[passwordName]`
        </ResponseField>
        <ResponseField name="inputPrefix" type="string">
          Default prefix for the authentication input field
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="playground" type="object">
      Configurations for the API playground.
      
      <Expandable title="Playground">
        <ResponseField name="mode" type="'show' | 'simple' | 'hide'" default="show">
          Whether the playground is showing, hidden, or only displaying the endpoint with no interactivity
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="maintainOrder" type="boolean">
      Ensures key ordering in OpenAPI pages matches the OpenAPI file definition.
      
      <Warning>This will soon be enabled by default.</Warning>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  URL(s) or relative path(s) pointing to your OpenAPI specification file.
  
  <CodeGroup>
    ```json Absolute URL
    "openapi": "https://example.com/openapi.json"
    ```
    
    ```json Relative Path
    "openapi": "/openapi.json"
    ```
    
    ```json Multiple Files
    "openapi": [
      "https://example.com/openapi1.json",
      "/openapi2.json",
      "/openapi3.json"
    ]
    ```
  </CodeGroup>
</ResponseField>

### User Feedback

<ResponseField name="feedback" type="object">
  Enable feedback buttons for users to suggest edits or raise issues.
  
  <Expandable title="Feedback Configuration">
    <ResponseField name="suggestEdit" type="boolean" default={false}>
      Enables a button allowing users to suggest edits via pull requests
    </ResponseField>
    
    <ResponseField name="raiseIssue" type="boolean" default={false}>
      Enables a button allowing users to raise documentation issues
    </ResponseField>
  </Expandable>

  ```json
  "feedback": {
    "suggestEdit": true,
    "raiseIssue": true
  }
  ```
</ResponseField>

### Contextual Actions

<ResponseField name="contextual" type="object">
  Configure contextual actions available when selecting text.
  
  <Expandable title="Contextual Options">
    <ResponseField name="options" type="array">
      Array of enabled contextual actions. Available options: `copy`, `view`, `chatgpt`, `claude`, `perplexity`, `mcp`, `cursor`, `vscode`
    </ResponseField>
  </Expandable>

  ```json Example from Mint Starter Kit
  "contextual": {
    "options": [
      "copy",
      "view",
      "chatgpt",
      "claude",
      "perplexity",
      "mcp",
      "cursor",
      "vscode"
    ]
  }
  ```
</ResponseField>

### Versioning

<ResponseField name="versions" type="string[]">
  Array of version names. Use this to show different versions of docs with a dropdown in the navigation bar.

  ```json
  "versions": ["v2.0", "v1.0"]
  ```
</ResponseField>

## Complete Example

Here's the complete `docs.json` from the Mint Starter Kit:

```json
{
  "$schema": "https://mintlify.com/docs.json",
  "theme": "mint",
  "name": "Mint Starter Kit",
  "colors": {
    "primary": "#16A34A",
    "light": "#07C983",
    "dark": "#15803D"
  },
  "favicon": "/favicon.svg",
  "logo": {
    "light": "/logo/light.svg",
    "dark": "/logo/dark.svg"
  },
  "navbar": {
    "links": [
      {
        "label": "Support",
        "href": "mailto:hi@mintlify.com"
      }
    ],
    "primary": {
      "type": "button",
      "label": "Dashboard",
      "href": "https://dashboard.mintlify.com"
    }
  },
  "footer": {
    "socials": {
      "x": "https://x.com/mintlify",
      "github": "https://github.com/mintlify",
      "linkedin": "https://linkedin.com/company/mintlify"
    }
  }
}
```
